<script>

// Open editor with category name function openEditor(category) {

document.getElementById('editorTitle').innerText = category; document.getElementById('editorContent').value = localStorage.getItem(category) || ''; document.getElementById('editorModal').style.display = 'block';

}


// Save content to localStorage function saveContent() {

const category = document.getElementById('editorTitle').innerText; const content = document.getElementById('editorContent').value; localStorage.setItem(category, content);

alert('Content saved!');

}


// Close editor

function closeEditor() { document.getElementById('editorModal').style.display = 'none';

}


// Add new category box function addCategory() {

const newCategory = prompt('Enter the new category name:'); if (newCategory) {

const categoryBox = document.createElement('div'); categoryBox.className = 'category-box'; categoryBox.textContent = newCategory; categoryBox.onclick = () => openEditor(newCategory);

document.querySelector('.container').appendChild(categoryBox);

}

}

</script>
